## FIX CODE
```c++
#include <stdio.h>

int main() {
    char name[50] ;
    int  age = 0 ;
    printf( "Enter your name : " ) ;
    scanf( "%s", name ) ;
    printf( "Enter your age : " ) ;
    scanf( "%d", &age ) ;
    printf( "------------------\n" ) ;
    printf( "Hello %s \n", name ) ; 
    printf( "Age = %d\n", age ) ; 
    
}
```
